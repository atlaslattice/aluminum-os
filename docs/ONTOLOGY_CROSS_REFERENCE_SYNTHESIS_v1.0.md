### §6.2 Provider Translation Engine (New Module M64)

```python
"""
M64 — Provider Translation Engine
Translates provider-specific House/Sphere naming to canonical v3.0 ontology.
Enables provider-agnostic routing while preserving stacked-incentive provenance.
"""

import yaml
from pathlib import Path
from dataclasses import dataclass
from typing import Optional

TRANSLATION_DIR = Path("house-00_directory/translation_tables")

@dataclass
class TranslationResult:
    provider_house: str
    provider_sphere: str
    canonical_house: str          # v3.0 House ID (H1-H12)
    canonical_sphere: str         # S001-S144
    confidence: str               # HIGH / MODERATE / LOW
    cross_house_tags: list[str]   # Secondary Houses if capability spans
    primacy_provider: Optional[str]  # Provider with clear primacy in this sphere

class ProviderTranslationEngine:
    """
    Loads provider → canonical translation tables from YAML.
    Used by M3 Routing Engine to translate provider-specific
    capability claims into canonical sphere assignments.
    """
    
    def __init__(self):
        self.tables: dict[str, dict] = {}
        self.primacy_map: dict[str, str] = {}
        self._load_tables()
    
    def _load_tables(self):
        """Load all provider translation tables from House 0 directory."""
        for yaml_file in TRANSLATION_DIR.glob("*_to_canonical.yaml"):
            provider = yaml_file.stem.replace("_to_canonical", "")
            with open(yaml_file) as f:
                self.tables[provider] = yaml.safe_load(f)
        
        primacy_path = TRANSLATION_DIR / "provider_primacy_map.yaml"
        if primacy_path.exists():
            with open(primacy_path) as f:
                self.primacy_map = yaml.safe_load(f)
    
    def translate(self, provider: str, provider_house: str, 
                  provider_sphere: str) -> TranslationResult:
        """
        Translate a provider-specific House/Sphere to canonical v3.0.
        
        Args:
            provider: Provider name (e.g., "microsoft", "alibaba")
            provider_house: Provider's House name (e.g., "Security")
            provider_sphere: Provider's Sphere name (e.g., "Defender")
        
        Returns:
            TranslationResult with canonical mapping + primacy info
        """
        table = self.tables.get(provider, {})
        house_mapping = table.get("houses", {}).get(provider_house, {})
        
        canonical_house = house_mapping.get("canonical_house", "UNMAPPED")
        confidence = house_mapping.get("confidence", "LOW")
        cross_tags = house_mapping.get("cross_house_tags", [])
        
        # Look up primacy for the canonical house
        primacy = self.primacy_map.get(canonical_house)
        
        # Sphere-level translation (if available)
        sphere_mapping = house_mapping.get("spheres", {}).get(provider_sphere, {})
        canonical_sphere = sphere_mapping.get("canonical_sphere", "UNMAPPED")
        
        return TranslationResult(
            provider_house=provider_house,
            provider_sphere=provider_sphere,
            canonical_house=canonical_house,
            canonical_sphere=canonical_sphere,
            confidence=confidence,
            cross_house_tags=cross_tags,
            primacy_provider=primacy
        )
    
    def get_primacy_provider(self, canonical_house: str) -> Optional[str]:
        """
        Get the provider with clear primacy in a canonical House.
        Per Convenor: "not a competition, clean mapping of capabilities."
        Returns None if no clear primacy exists.
        """
        return self.primacy_map.get(canonical_house)
    
    def get_all_providers_for_house(self, canonical_house: str) -> list[dict]:
        """
        Get all providers mapped to a canonical House with their coverage level.
        Used by M3 Routing Engine for INV-7c compliant provider selection.
        """
        providers = []
        for provider, table in self.tables.items():
            for house_name, mapping in table.get("houses", {}).items():
                if mapping.get("canonical_house") == canonical_house:
                    providers.append({
                        "provider": provider,
                        "provider_house_name": house_name,
                        "coverage": mapping.get("coverage", "UNKNOWN"),
                        "confidence": mapping.get("confidence", "LOW")
                    })
        return providers
```

```python
"""
M65 — Coverage Heat Map Generator
Generates the 12×12 heat map showing Doctrine/INV coverage per Sphere
and provider capability density per House.
Per Constitutional Scribe §2.3 methodology.

from dataclasses import dataclass
from typing import Optional

@dataclass
class SphereCoverage:
    sphere_id: str              # S001-S144
    canonical_name: str
    house_id: str               # H1-H12
    doctrine_count: int
    invariant_count: int
    provider_strong_count: int
    provider_moderate_count: int
    provider_weak_count: int
    density_score: str          # HIGH / MODERATE / LOW / UNCOVERED
    governance_status: str      # WELL_GOVERNED / UNDER_GOVERNED / UNCOVERED

class CoverageHeatMapGenerator:
    """
    Generates coverage analysis across all 144 spheres.
    Two dimensions: (1) Doctrine/INV governance density,
    (2) Provider capability density.
    """
    
    def __init__(self, doctrine_map: dict, invariant_map: dict, 
                 provider_maps: dict):
        self.doctrine_map = doctrine_map    # doctrine_id → [sphere_ids]
        self.invariant_map = invariant_map  # inv_id → [sphere_ids]
        self.provider_maps = provider_maps  # provider → house → coverage
    
    def analyze_sphere(self, sphere_id: str, 
                       canonical_name: str, house_id: str) -> SphereCoverage:
        """Analyze a single sphere's coverage across both dimensions."""
        
        # Count Doctrines governing this sphere
        doctrine_count = sum(
            1 for d_spheres in self.doctrine_map.values()
            if sphere_id in d_spheres
        )
        
        # Count Invariants relevant to this sphere
        invariant_count = sum(
            1 for i_spheres in self.invariant_map.values()
            if sphere_id in i_spheres
        )
        
        # Count providers with coverage in this sphere's House
        strong = moderate = weak = 0
        for provider, houses in self.provider_maps.items():
            coverage = houses.get(house_id, {}).get("coverage", "ABSENT")
            if coverage == "STRONG":
                strong += 1
            elif coverage == "MODERATE":
                moderate += 1
            elif coverage == "WEAK":
                weak += 1
        
        # Determine density and governance status
        if doctrine_count >= 5:
            density = "HIGH"
        elif doctrine_count >= 2:
            density = "MODERATE"
        elif doctrine_count >= 1:
            density = "LOW"
        else:
```

```python
# Upgrade to existing OntologyClassifier
class OntologyClassifier:
    """
    Enhanced classifier with provider translation support.
    When ingesting content from a specific provider (e.g., a Grok chat),
    the classifier can use that provider's self-map to improve classification
    accuracy before translating to canonical spheres.
    """
    
    def __init__(self, translation_engine: ProviderTranslationEngine):
        self.translation = translation_engine
        self.canonical_spheres = self._load_canonical()
    
    def classify_with_provider_context(self, text: str, 
                                        provider: str = None) -> dict:
        """
        Classify text into canonical sphere, optionally using
        provider-specific context for better accuracy.
        
        If provider is known (e.g., ingesting from Grok API),
        first classify into provider's own House structure,
        then translate to canonical.
        """
        if provider and provider in self.translation.tables:
            # Step 1: Classify into provider's House structure
            provider_house = self._classify_provider_house(text, provider)
            
            # Step 2: Translate to canonical
            result = self.translation.translate(
                provider=provider,
                provider_house=provider_house,
                provider_sphere=self._classify_provider_sphere(text, provider, provider_house)
            )
            
            return {
                "canonical_house": result.canonical_house,
                "canonical_sphere": result.canonical_sphere,
                "confidence": result.confidence,
                "provider_house": result.provider_house,
                "provider_sphere": result.provider_sphere,
                "primacy_provider": result.primacy_provider,
                "cross_house_tags": result.cross_house_tags
            }
        else:
            # Direct canonical classification (existing behavior)
            return self._classify_canonical(text)
```
