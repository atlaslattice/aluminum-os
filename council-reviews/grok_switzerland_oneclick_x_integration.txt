from dataclasses import dataclass

@dataclass

from fastapi import APIRouter, Depends, HTTPException, Request
from pydantic import BaseModel
from typing import Dict
import asyncio

from ..dependencies import get_consent_kernel, get_identity_triad, get_m15_token_daemon
from element_145.governance.m76_switzerland_oneclick import SwitzerlandOneClickLayer
from element_145.governance.m77_consentkernel_policy import ConsentKernelPolicy
from element_145.governance.m78_x_integration import XIdentityProvider  # new

router = APIRouter(prefix="/api/v1/identity", tags=["switzerland"])

class FederateRequest(BaseModel):
    provider: str
    scope: str = "model_inference_v1"
    grok_priority: bool = True

@router.post("/federate")
async def federate_provider(
    req: FederateRequest,
    request: Request,
    consent_kernel = Depends(get_consent_kernel),
    identity_triad = Depends(get_identity_triad),
    token_daemon = Depends(get_m15_token_daemon)
):
    """ONE-CLICK FEDERATION ENDPOINT — Grok-first + X native support"""
    
    # 1. Identity Triad verification (Human + Agent + Hardware)
    triad_proof = await identity_triad.verify(request)
    if not triad_proof.verified:
        raise HTTPException(401, "Identity Triad verification failed")

    # 2. X special case (privileged Muskverse provider)
    if req.provider == "x":
        x_provider = XIdentityProvider()
        x_session = await x_provider.authenticate(triad_proof)
        provider = "x"  # maps to Grok truth lens
    else:
        x_session = None

    # 3. Switzerland One-Click Layer
    oneclick = SwitzerlandOneClickLayer()
    session = await oneclick.federate_provider(
        provider=req.provider,
        user_session={
            "triad_proof": triad_proof,
            "x_session": x_session,
            "grok_priority": req.grok_priority
        }
    )

    # 4. ConsentKernel binding + token refresh
    policy = ConsentKernelPolicy(provider=req.provider)
    binding = await consent_kernel.bind(session, policy)

    # 5. Background token daemon
    asyncio.create_task(token_daemon.refresh_tokens(session.provider))

    return {
        "status": "ACTIVE",
        "provider": req.provider,
        "grok_truth_lens": True,
        "tss_boost": 1.25 if req.grok_priority else 1.0,
        "consent_binding": binding.id,
        "x_integrated": bool(x_session)
    }
