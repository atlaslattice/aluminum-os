# 🧠 1. Provider Registry

## `core/provider_registry.py`

```python
class ProviderRegistry:
    def __init__(self, vault, consent_kernel):
        self.vault = vault
        self.consent_kernel = consent_kernel
        self.providers = {}

    def register(self, name, credential_key, scopes):
        self.providers[name] = {
            "credential_key": credential_key,
            "scopes": scopes,
            "active": False
        }

    def activate(self, name):
        if name not in self.providers:
            raise ValueError("Provider not registered")

        if not self.consent_kernel.check(name):
            raise Exception("Consent not granted")

        credential = self.vault.get_secret(self.providers[name]["credential_key"])

        if not credential:
            raise Exception("Missing credential")

        self.providers[name]["active"] = True
        return True

    def get_active(self):
        return [k for k, v in self.providers.items() if v["active"]]
```

```python
class SecureVault:
    """
    Abstract vault layer.
    Replace backend with:
    - Azure Key Vault

    def __init__(self):
        self._memory = {}

    def store_secret(self, key, value):
        self._memory[key] = value

    def get_secret(self, key):
        return self._memory.get(key)

    def unlock(self, auth_token):
```

# 🧾 3. ConsentKernel

## `core/consent_kernel.py`

```python
class ConsentKernel:
    def __init__(self):
        self.consents = {}

    def grant(self, provider):
        self.consents[provider] = True

    def revoke(self, provider):
        self.consents[provider] = False

    def check(self, provider):
        return self.consents.get(provider, False)
```

# 🧾 4. Audit Chain (provenance)

## `core/audit_chain.py`

```python
import hashlib

class AuditChain:
    def __init__(self):
        self.chain = []

    def append(self, record):
        prev = self.chain[-1]["hash"] if self.chain else "GENESIS"
        raw = str(record) + prev
        h = hashlib.sha256(raw.encode()).hexdigest()

        self.chain.append({
            "record": record,
            "hash": h,
            "prev": prev
        })
```

## OpenAI-compatible client

## `providers/openai_client.py`

```python
import requests

class OpenAIClient:
    def __init__(self, api_key):
        self.api_key = api_key
        self.endpoint = "https://api.openai.com/v1/chat/completions"

    def call(self, prompt):
        r = requests.post(
            self.endpoint,
            headers={
                "Authorization": f"Bearer {self.api_key}"
            },
            json={
                "model": "gpt-4o-mini",
                "messages": [{"role": "user", "content": prompt}]
            }
        )
        return r.json()
```

# 🔁 6. Boot Loader (this is your “root experience”)

## `boot/boot_loader.py`

```python
def boot_system(vault, registry):
    # Step 1 — unlock vault
    if not vault.unlock("VALID_USER_SESSION"):
        raise Exception("Vault unlock failed")

    # Step 2 — activate providers with consent
    for provider in registry.providers:
        try:
            registry.activate(provider)
        except Exception:
            continue

    return registry.get_active()
```

```python
from fastapi import FastAPI
from core.provider_registry import ProviderRegistry
from core.vault import SecureVault
from core.consent_kernel import ConsentKernel

app = FastAPI()

vault = SecureVault()
consent = ConsentKernel()
registry = ProviderRegistry(vault, consent)

# seed example
vault.store_secret("openai_key", "sk-xxxx")
registry.register("openai", "openai_key", ["inference"])
consent.grant("openai")


@app.post("/provider/activate")
def activate_provider(name: str):
    registry.activate(name)
    return {"status": "ACTIVE", "provider": name}


@app.get("/providers/active")
def active_providers():
    return registry.get_active()
```
